📄 Sample Project Proposal 
The Blog Website System is a web-based application that allows users to create, publish, and manage blog posts online. Registered users can write articles, edit or delete their posts, and comment on other users’ blogs. Visitors can read published posts without registration. An administrator manages users, categories, and content moderation. The system aims to provide an easy-to-use platform for sharing knowledge and ideas.

                                                                                 * Pseudocode

1. User Login
Algorithm: UserLogin
Input: username, password
Output: login success or failure
BEGIN
    Read username and password from input
    IF username exists in database THEN
        IF password matches database record THEN
            Create user session
            Display "Login Successful"
        ELSE
            Display "Incorrect Password"
        ENDIF
    ELSE
        Display "User does not exist"
    ENDIF
END
2. Create Blog Post
Algorithm: CreatePost
Input: postTitle, postContent, authorId
Output: success/failure message
BEGIN
    IF user is authenticated THEN
        IF postTitle and postContent are not empty THEN
            Insert new post into Posts table:
                - title = postTitle
                - content = postContent
                - authorId = authorId
                - createdAt = current timestamp
            Display "Post Created Successfully"
        ELSE
            Display "Title or Content cannot be empty"
        ENDIF
    ELSE
        Display "User not authenticated"
    ENDIF
END
3. Add Comment
Algorithm: AddComment
Input: postId, userId, commentContent
Output: success/failure message
BEGIN
    IF user is authenticated THEN
        IF commentContent is not empty THEN
            Insert new comment into Comments table:
                - postId = postId
                - userId = userId
                - content = commentContent
                - createdAt = current timestamp
            Display "Comment Added Successfully"
        ELSE
            Display "Comment cannot be empty"
        ENDIF
    ELSE
        Display "User not authenticated"
    ENDIF
END
4. Fetch All Posts
Algorithm: GetAllPosts
Input: none
Output: list of posts
BEGIN
    Query Posts table
    FOR each post in result DO
        Fetch post title, content, author
        Fetch comments for post
        Display post and comments
    END FOR
END
5. Admin Moderation of Post or Comment
Algorithm: ModerateContent
Input: contentId, actionType (delete/block)
Output: success/failure message
BEGIN
    IF user.role = 'Admin' THEN
        IF actionType = 'deletePost' THEN
            Delete post from Posts table where postId = contentId
            Display "Post Deleted Successfully"
        ELSE IF actionType = 'deleteComment' THEN
            Delete comment from Comments table where commentId = contentId
            Display "Comment Deleted Successfully"
        ENDIF
    ELSE
        Display "Permission Denied"
    ENDIF
END

-- DATABASE: Blog Website
CREATE DATABASE IF NOT EXISTS blog_website;
USE blog_website;
-- 1. Users Table
CREATE TABLE IF NOT EXISTS users (
    id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(50) NOT NULL UNIQUE,
    email VARCHAR(100) NOT NULL UNIQUE,
    password VARCHAR(255) NOT NULL,
    profile_image VARCHAR(255) DEFAULT NULL,
    bio TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);
-- 2. Posts Table
CREATE TABLE IF NOT EXISTS posts (
    id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    user_id INT UNSIGNED NOT NULL,
    title VARCHAR(255) NOT NULL,
    content TEXT NOT NULL,
    featured_image VARCHAR(255) DEFAULT NULL,
    status ENUM('draft', 'published') DEFAULT 'draft',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
);
-- 3. Comments Table
CREATE TABLE IF NOT EXISTS comments (
    id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    post_id INT UNSIGNED NOT NULL,
    user_id INT UNSIGNED NOT NULL,
    content TEXT NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (post_id) REFERENCES posts(id) ON DELETE CASCADE,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
);
-- 4. Categories Table
CREATE TABLE IF NOT EXISTS categories (
    id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(50) NOT NULL UNIQUE,
    description TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);
-- 5. Tags Table
           CREATE TABLE IF NOT EXISTS tags (
    id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(50) NOT NULL UNIQUE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);


-- 6. Post_Tags Table (Many-to-Many)
-- =====================================
CREATE TABLE IF NOT EXISTS post_tags (
    post_id INT UNSIGNED NOT NULL,
    tag_id INT UNSIGNED NOT NULL,
    PRIMARY KEY (post_id, tag_id),
    FOREIGN KEY (post_id) REFERENCES posts(id) ON DELETE CASCADE,
    FOREIGN KEY (tag_id) REFERENCES tags(id) ON DELETE CASCADE
);
-- 7. Likes Table
CREATE TABLE IF NOT EXISTS likes (
    id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    post_id INT UNSIGNED NOT NULL,
    user_id INT UNSIGNED NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (post_id) REFERENCES posts(id) ON DELETE CASCADE,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE,
    UNIQUE KEY unique_like (post_id, user_id));
Error handling
DELIMITER $$
CREATE TRIGGER trg_posts_before_insert
BEFORE INSERT ON posts
FOR EACH ROW
BEGIN
    IF CHAR_LENGTH(NEW.title) < 5 THEN
        SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Post title too short (min 5 characters)';
    END IF;
    IF CHAR_LENGTH(NEW.content) < 10 THEN
        SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Post content too short (min 10 characters)';
    END IF;
END$$
DELIMITER ;
-- Prevent empty comment
DELIMITER $$
CREATE TRIGGER trg_comments_before_insert
BEFORE INSERT ON comments
FOR EACH ROW
BEGIN
    IF CHAR_LENGTH(NEW.content) = 0 THEN
        SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Comment cannot be empty';
    END IF;
END$$
DELIMITER ;
